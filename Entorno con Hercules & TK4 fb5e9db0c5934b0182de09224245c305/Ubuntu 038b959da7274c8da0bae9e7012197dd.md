[**TK4**](https://wotho.ethz.ch/tk4-/)

```bash
# Creamos una carpeta oculta dentro del home
mkdir ~/.mvs && cd ~/.mvs/

# Descargamos TK4
wget --no-check-certificate https://wotho.ethz.ch/tk4-/tk4-_v1.00_current.zip

# Descomprimir el archivo zip
unzip tk4-_v1.00_current.zip

# Ahora entramos a la carpeta conf y modificamos el archivo tk4-.cnf
NUMCPU ${NUMCPU:=2} # Aumentamos de 1 a 2
MAXCPU ${MAXCPU:=2} # Aumentamos de 1 a 2
TZOFFSET +0500 # Definir nuestra zona horaria

# Vamos a la carpeta unattended y ejecutamos el script
cd .. && cd unattended && ./set_console_mode

# En este punto ya podemos ejecutar MVS, podemos crear un alias para ejecutarlo
alias mvs="cd ~/.mvs/ && ./mvs"

# Ejecutar MVS
mvs
```
